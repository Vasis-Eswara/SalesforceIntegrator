✅ 1. Update your schema_view.html (Jinja2 Template)
Insert the following where you render the “Salesforce Objects” section:
<div class="mb-3">
  <h5 class="text-white">Salesforce Objects</h5>
  <input type="text" id="objectSearch" class="form-control bg-dark text-light" placeholder="Search objects...">
</div>
 
<div class="object-list" style="max-height: 500px; overflow-y: auto;">

✅ 2. Add JavaScript: static/js/schema_search.js
document.addEventListener("DOMContentLoaded", function () {
  const searchInput = document.getElementById("objectSearch");
  const items = document.querySelectorAll(".object-item");
  const noResults = document.getElementById("noResults");
 
  searchInput.addEventListener("input", function () {
    const query = this.value.toLowerCase();
    let visibleCount = 0;
 
    items.forEach(item => {
      const text = item.textContent.toLowerCase();
      const match = text.includes(query);
      item.style.display = match ? "block" : "none";
      if (match) visibleCount++;
    });
 
    noResults.style.display = visibleCount === 0 ? "block" : "none";
  });
});
✅ 3. Load the JS in your template
Add this at the bottom of your schema_view.html file:
<script src="{{ url_for('static', filename='js/schema_search.js') }}"></script>
